        // Download all slides at once helper setup loop
        function downloadAllPins() {
            let delay = 0;
            const exportWrapper = document.getElementById('export-hidden-container');

            for (let i = 0; i < pinSlidesData.length; i++) {
                setTimeout(() => {
                    exportWrapper.innerHTML = `
                        <div id="temp-canvas-export-${i}" style="width: 1000px; height: 1500px; font-size: 2.2rem; line-height: 1.5;">
                            ${pinSlidesData[i]}
                        </div>
                    `;
                    const node = document.getElementById(`temp-canvas-export-${i}`);
                    html2canvas(node, {
                        scale: 2,
                        useCORS: true,
                        allowTaint: true
                    }).then(canvas => {
                        const imgData = canvas.toDataURL('image/png');
                        const link = document.createElement('a');
                        link.download = `Akhil-Baburaj-Pinterest-Slide-${i + 1}.png`;
                        link.href = imgData;
                        link.click();
                        exportWrapper.innerHTML = '';
                    });
                }, delay);
                delay += 1000; // 1 second interval to prevent blocking multiple downloads
            }
        }

        function setPlannerGoal(goal) {
            const buttons = document.querySelectorAll('.planner-goal-btn');
            buttons.forEach(btn => btn.classList.remove('border-mustard', 'bg-forestDark/80', 'active'));
            
            const activeBtn = document.getElementById(`btn-${goal}`);
            if (activeBtn) {

            }

            renderPlannerOutput();
        }

        function renderPlannerOutput() {

            document.getElementById('blueprintTitle').innerText = blueprint.title;
            document.getElementById('blueprintDesc').innerText = blueprint.desc;
            document.getElementById('blueprintMetrics').innerText = blueprint.metrics;

            const roadmapElement = document.getElementById('blueprintRoadmap');

                    <span class="text-mustard font-bold">✓</span>
                    <span class="text-cream/80">${step}</span>
                `;
                roadmapElement.appendChild(li);
            });
        }
